4. JavaScript for Clock and Date-Time Page
In the assets/js/script.js file, you can add the following code to make the clock and date/time interactive:

javascript
Copy
// Function to update time on the clock page
function updateClock() {
    const timeElement = document.getElementById('time');
    const currentTime = new Date();
    timeElement.textContent = currentTime.toLocaleTimeString();
}

// Function to update date/time on the Date & Time page
function updateDateTime() {
    const dateTimeElement = document.getElementById('datetime');
    const currentDateTime = new Date();
    dateTimeElement.textContent = currentDateTime.toLocaleString();
}

// Update clock every second
setInterval(updateClock, 1000);

// Update date & time every second
setInterval(updateDateTime, 1000);
